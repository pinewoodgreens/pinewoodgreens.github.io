  Foreground: #9BB89E (--pine-light)
  Background: #1A3820 (--pine-deep)

If node is available, run:
```bash
node -e "
const hex = h => [parseInt(h.slice(1,3),16)/255, parseInt(h.slice(3,5),16)/255, parseInt(h.slice(5,7),16)/255].map(c => c <= 0.04045 ? c/12.92 : Math.pow((c+0.055)/1.055,2.4));
const lum = ([r,g,b]) => 0.2126*r + 0.7152*g + 0.0722*b;
const ratio = (a,b) => { const [l1,l2] = [lum(hex(a)),lum(hex(b))].sort((a,b)=>b-a); return ((l1+0.05)/(l2+0.05)).toFixed(2); };
console.log('Contrast ratio:', ratio('#9BB89E','#1A3820'), ':1');
console.log('WCAG AA pass (4.5:1):', ratio('#9BB89E','#1A3820') >= 4.5 ? 'YES' : 'NO');
"
```
